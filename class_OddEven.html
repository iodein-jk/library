import { class_OddEven } from "./module/class_OddEven.js"
window.addEventListener('DOMContentLoaded', function(){
  class_OddEven({
    select: '.chronology-Item',
    type: 0
  })
})

Parameter

Key Description Default Required
select 取得するHTMLのclass属性を指定 Required
type クラス付与を[奇数][偶数]のどちらにすると指定する。
0=[偶数]
1=[奇数]		 0
