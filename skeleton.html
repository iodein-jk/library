import { skeleton } from "./module/skeleton.js"
window.addEventListener("DOMContentLoaded", function(){
  skeleton({
    id:'skeleton', 
    clip: 60
  })
})

Parameter

Key Description Default Required
id 対象のid属性を指定 無 Required
clip スケルトンのエリア幅を指定 0 Required
